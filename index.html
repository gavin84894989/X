import * as THREE from 'https://cdn.skypack.dev/three@0.132.2'; import { GLTFLoader } from 'https://cdn.skypack.dev/three@0.132.2/examples/jsm/loaders/GLTFLoader.js'; // --- A. 基本設定 --- const scene = new THREE.Scene(); const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000); const renderer = new THREE.WebGLRenderer({ antialias: true }); renderer.setSize(window.innerWidth, window.innerHeight); document.body.appendChild(renderer.domElement); const raycaster = new THREE.Raycaster(); const mouse = new THREE.Vector2(); // --- B. 資源路徑 (建議全放在 GitHub 同一資料夾下) --- const BASE_URL = 'https://raw.githubusercontent.com/gavin84894989/X/main/'; const MODEL_PATH = BASE_URL + 'scene.glb'; const DOLL_IMAGE_PATH = BASE_URL + 'doll.png'; // 請準備這張圖 const SOUND_PATH = BASE_URL + 'cry.mp3'; // 請準備這個檔 // --- C. 手電筒 --- const flashlight = new THREE.SpotLight(0xffffff, 20); flashlight.angle = Math.PI / 8; flashlight.penumbra = 0.5; flashlight.distance = 25; scene.add(flashlight); scene.add(flashlight.target); // --- D. 載入車廂 --- const loader = new GLTFLoader(); loader.load(MODEL_PATH, (gltf) => { scene.add(gltf.scene); gltf.scene.traverse(n => { if(n.isMesh) n.material.side = THREE.DoubleSide; }); }); // --- E. 平面娃娃 (Sprite) --- // Sprite 是一種永遠會面對鏡頭的平面圖，適合做驚悚遊戲的鬼魂 const textureLoader = new THREE.TextureLoader(); const dollMap = textureLoader.load(DOLL_IMAGE_PATH); const dollMaterial = new THREE.SpriteMaterial({ map: dollMap, transparent: true }); const doll = new THREE.Sprite(dollMaterial); doll.scale.set(0.5, 0.8, 1); // 調整娃娃的大小比例 doll.position.set(0, -0.4, -5); doll.visible = false; scene.add(doll); // --- F. 遊戲邏輯 --- let shineCount = 0; let isLookingAtDoll = false; let hasTriggered = false; let gameEnded = false; const crySound = new Audio(SOUND_PATH); crySound.loop = true; window.addEventListener('mousemove', (e) => { mouse.x = (e.clientX / window.innerWidth) * 2 - 1; mouse.y = -(e.clientY / window.innerHeight) * 2 + 1; flashlight.target.position.set(mouse.x * 10, mouse.y * 10, -10); flashlight.position.copy(camera.position); }); window.addEventListener('click', () => { if (doll.visible && !gameEnded) { raycaster.setFromCamera(mouse, camera); if (raycaster.intersectObject(doll).length > 0) showOptions(); } }); function animate() { requestAnimationFrame(animate); if (!gameEnded) { // 偵測是否照到娃娃 raycaster.setFromCamera(mouse, camera); if (raycaster.intersectObject(doll).length > 0) { if (!isLookingAtDoll) { isLookingAtDoll = true; shineCount++; if (shineCount >= 3) triggerEvent(); } } else { isLookingAtDoll = false; } flashlight.intensity = 5 + Math.random() * 15; } renderer.render(scene, camera); } animate(); function triggerEvent() { hasTriggered = true; doll.visible = true; crySound.play().catch(() => console.log("等待點擊播放")); const msg = document.createElement('div'); msg.style = "position:fixed; top:20px; width:100%; text-align:center; color:white; font-size:24px; text-shadow:2px 2px 5px black;"; msg.innerHTML = "咦？什麼時候出現這個東西的？"; document.body.appendChild(msg); } function showOptions() { const menu = document.createElement('div'); menu.style = "position:fixed; top:50%; left:50%; transform:translate(-50%, -50%); background:rgba(0,0,0,0.9); padding:30px; border:1px solid red; color:white; text-align:center; z-index:100;"; menu.innerHTML = `

你要怎麼處理這個玩偶？





`; document.body.appendChild(menu); } window.endGame = function(msg) { gameEnded = true; crySound.pause(); document.body.innerHTML = `
${msg}
`; };
